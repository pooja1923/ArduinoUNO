#include<LiquidCrystal.h>

LiquidCrystal lcd(8,9,10,11,12,13);
const int lm35Pin= A0;

void setup(){
  Serial.begin(9600);
  lcd.begin(16,2);
  lcd.print("LM35 Test");
  delay(2000);
  lcd.clear();
}

void loop(){
  int sensorValue=analogRead(lm35Pin);
  float voltage=sensorValue*(5.05/1023.0);
  float tempC=voltage*100.0;
  float tempF=(tempC*9.0/5.0)+32.0;
  
  Serial.print("Temp:");
  Serial.print(tempC);
  Serial.print("C");
  Serial.print(tempF);
  Serial.print("F");
  
  lcd.setCursor(0,0);
  lcd.print("C:");
  lcd.print(tempC);
  lcd.print(" ");
  
  lcd.setCursor(0,1);
  lcd.print("F:");
  lcd.print(tempF);
  lcd.print(" ");
  
  delay(1000);
}
